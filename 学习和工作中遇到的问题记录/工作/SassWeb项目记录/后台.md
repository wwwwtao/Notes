// https://umijs.org/config/
import os from 'os'
import pageRoutes from './router.config'
import webpackPlugin from './plugin.config'
import defaultSettings from '../src/defaultSettings'

let domainName = ''
switch (process.env.SERVICE_TYPE) {
case 'pro':
  // domainName = 'http://passportapi.xiaokeduo.net';
  domainName = 'https://supplierapi.yunquanlaile.com'
  break
case 'fat':
  domainName = 'https://supplierapi-fat.xiaokeduo.com'
  break
case 'dev':
  domainName = 'https://supplierapi-local.xiaokeduo.com'
  break
case 'uat':
  domainName = 'https://supplierapi-uat.xiaokeduo.com'
  break
default:
  domainName = ''
  break
}

export default {
  // disableCSSSourceMap: process.env.SERVICE_TYPE==='pro'?true:false,
  //devtool: process.env.SERVICE_TYPE==='pro'?"nosources-source-map":"source-map",
  publicPath: process.env.SERVICE_TYPE === 'pro' ? 'https://img.xiaokeduo.com/xkdnewyun/supplier/' : '/',
  // add for transfer to umi
  plugins: [
    [
      'umi-plugin-react',
      {
        antd: true,  //整合 antd 组件库。
        dva: {  //整合 dva 数据流。
          hmr: true,
        },
        targets: {
          ie: 11,
        },
        locale: {  //国际化插件，用于解决 i18n 问题
          enable: true, // default false
          default: 'zh-CN', // default zh-CN
          baseNavigator: true, // default true, when it is true, will use `navigator.language` overwrite default
        },
        dynamicImport: { //是否启用按需加载，即是否把构建产物进行拆分，在需要的时候下载额外的 JS 再执行。
          loadingComponent: './components/pageloading/index',  // 加载时的展示
        },
        ...(!process.env.TEST && os.platform() === 'darwin'
          ? {
            dll: {
              include: ['dva', 'dva/router', 'dva/saga', 'dva/fetch'],
              exclude: ['@babel/runtime'],
            },
            hardSource: true,
          }
          : {}),
      },
    ],
    [
      'umi-plugin-ga',
      {
        code: 'UA-72788897-6',
        judge: () => process.env.APP_TYPE === 'site',
      },
    ],
  ],
  targets: {
    ie: 11,
  },
  define: {
    APP_TYPE: process.env.APP_TYPE || '',
    DOMAIN: domainName,
    SERVICE_TYPE: process.env.SERVICE_TYPE || '',
  },
  // 路由配置
  routes: pageRoutes,
  // Theme for antd
  // https://ant.design/docs/react/customize-theme-cn
  theme: {
    'primary-color': defaultSettings.primaryColor,
  },
  externals: {  //设置哪些模块可以不被打包，通过 <script> 或其他方式引入，通常需要和 scripts 或 headScripts 配置同时使用。
    '@antv/data-set': 'DataSet',
  },
  // proxy: {
  //   '/server/api/': {
  //     target: 'https://preview.pro.ant.design/',
  //     changeOrigin: true,
  //     pathRewrite: { '^/server': '' },
  //   },
  // },
  proxy: {
    '/100000000001': {
      target: 'https://supplierapi-local.xiaokeduo.com/100000000001',
      changeOrigin: true,
      pathRewrite: {
        '^/100000000001': '',
      },
    },
  },
  ignoreMomentLocale: true,  //忽略 moment 的 locale 文件，用于减少尺寸。
  lessLoaderOptions: { //给 less-loader 的额外配置项。
    javascriptEnabled: true, //启用 js
  },
  disableRedirectHoist: true, //禁用 redirect 上提。
  cssLoaderOptions: {  //css-loader
    modules: true, //启用
    getLocalIdent: (context, localIdentName, localName) => {  //使用内置函数来生成类名
      if (
        context.resourcePath.includes('node_modules') ||
        context.resourcePath.includes('ant.design.pro.less') ||
        context.resourcePath.includes('global.less')
      ) {
        return localName
      }
      const match = context.resourcePath.match(/src(.*)/)
      if (match && match[1]) {
        const antdProPath = match[1].replace('.less', '')
        const arr = antdProPath
          .split('/')
          .map(a => a.replace(/([A-Z])/g, '-$1'))
          .map(a => a.toLowerCase())
        return `antd-pro${arr.join('-')}-${localName}`.replace(/--/g, '-')
      }
      return localName
    },
  },
  manifest: {  // 生成manifest
    name: 'ant-design-pro',
    background_color: '#FFF',
    description: 'An out-of-box UI solution for enterprise applications as a React boilerplate.',
    display: 'standalone',
    start_url: '/index.html',
    icons: [
      {
        src: '/favicon.png',
        sizes: '48x48',
        type: 'image/png',
      },
    ],
  },

  history: 'hash',
  chainWebpack: webpackPlugin,
  cssnano: {
    mergeRules: false,  //关闭通过选择器和重叠的属性/值对合并相邻规则
  },
  theme: './config/theme.js',  //主题色
}
